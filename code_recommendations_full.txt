def detect_branch() -> Optional[str]:
    """Detect the current Git branch.
        Current branch name or None if not in a Git repository
        repo_path = config.git_repo_path
        # Change to repo directory if needed
        original_cwd = None
        if repo_path != Path("."):
            original_cwd = Path.cwd()
            repo_path = repo_path.resolve()
            repo_path.chdir()
            # Try git command first
            result = subprocess.run(
                ["git", "branch", "--show-current"],
                capture_output=True,
                text=True,
                timeout=10
            if result.returncode == 0:
                branch = result.stdout.strip()
                if branch:
                    logger.debug(f"Detected branch via git: {branch}")
                    return branch
            # Fallback: parse HEAD file
            git_head_path = repo_path / ".git" / "HEAD"
            if git_head_path.exists():
                try:
                    with open(git_head_path, "r") as f:
                        head_content = f.read().strip()
                        if head_content.startswith("ref: refs/heads/"):
                            branch = head_content.replace("ref: refs/heads/", "")
                            logger.debug(f"Detected branch via HEAD file: {branch}")
                            return branch
                except (IOError, OSError) as e:
                    logger.debug(f"Could not read HEAD file: {e}")
            # Check for detached HEAD
                        if len(head_content) == 40 and all(c in "0123456789abcdef" for c in head_content):
                            logger.debug("Detected detached HEAD state")
                            return "detached"
                    logger.debug(f"Could not read HEAD file for detached check: {e}")
        finally:
            # Restore original directory if changed
            if original_cwd and original_cwd.exists():
                original_cwd.chdir()
        logger.debug("Could not detect current branch")
    except (subprocess.TimeoutExpired, subprocess.SubprocessError, FileNotFoundError) as e:
        logger.debug(f"Git command failed: {e}")
    except Exception as e:
        logger.warning(f"Unexpected error detecting branch: {e}")
